base64 [FILE]           #Imprime le résultat de l'encodage de FILE (par défaut 
                        #stdin) en base64 (RFC 4648).
                        #base64 est utilisé pour représenté avec seulement des
                        #caractères imprimables, soit [[:alnum:]+/=] l'ensemble
                        #des 256 valeurs possibles d'un octet, pour transport
                        #sur un media ne permettant pas le transport des 256
                        #valeurs possibles.
                        #Les newlines n'étant pas significatifs en base64, un
                        #newline est ajouté tous les LARGEUR caractères (par
                        #défaut 76).

-d                      #Imprime le résultat du décodage, et non l'inverse :
--decode                #FILE doit donc être encodé en base64.
-i                      #Avec -d, ne produit pas d'erreur si FILE contient des
--ignore-garbage        #caractères en dehors de [[:alnum:]+/=] : ils sont
                        #seulement ignorés.

-w NOMBRE               
--wraps=NOMBRE          #LARGEUR devient NOMBRE (inutile si -d).
