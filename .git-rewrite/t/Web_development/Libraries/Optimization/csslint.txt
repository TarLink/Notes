csslint FILE|DIR							#CSS linting (version 0.10.0)
															#Exit code 1 if errors, 0 if warnings or nothing. Def: warnings all.
															#If ./.csslintrc present, reads command line options from it (should be "--VAR", one per line or
															#not)
															#Can also override some rules per CSS file by including a CSS comment on top of CSS file:
															#  /*csslint RULE: BOOL[,RULE: BOOL]*/ (no space between /* and csslint)
															#Rules are below.
--exclude-list=FILE|DIR,...		#
--format=STR									#Reporter, among: text (def), compact, lint-xml, csslint-xml, checkstyle-xml, junit-xml
--quiet												#
--list-rules									#
--errors|warnings|
ignore=RULE,...								#Def: warnings all. If set, def becomes ignore all.

POSSIBLE MISTAKE ==>					#
known-properties							#CSS property exists
box-model											#width|height used together with border* or padding* and no box-sizing
empty-rules										#Empty CSS rule
duplicate-properties					#Same property in one CSS rule
display-property-grouping			#Any of:
															#  - display: inline with width|height
															#  - display: inline*|table-* with float
															#  - display: inline|table-* with margin*
															#  - display: block with vertical-align

BAD PRACTICE ==>							#
universal-selector						#*
important											#!important
import												#@import
ids														# #ID (prefer .class)
text-ident										#text-ident: -NUM
outline-none									#outline: none (bad for accessibility)
overqualified-elements				#TAG.CLASS, unless TAG2.CLASS also exist (decoupling HTML and CSS, and increasing reusability)
qualified-headings						#h1-h6 with another CSS selector (h1-h6 styling should be the same across the whole website).
unique-headings								#same h1-h6 selector across rules, unless with different selectors following it (same reason as
															#above)

SAVES BYTES ==>								#															
zero-units										#0px, 0em, 0%, etc. instead of just 0 (saves bytes)
duplicate-background-images		#2 rules using background[-image] with same URL
shorthand											#Use shorthand properties if all properties used (less bytes)

PERFORMANCE ==>								#
regex-selectors								#[ATTR~=VAL], [ATTR|=VAL], [ATTR^=VAL], [ATTR$=VAL], [ATTR*=VAL] (performance issues)
															#But [ATTR] and [ATTR=VAL] are ok
unqualified-attributes				#[ATTR] or [ATTR=VAL] followed by no other selector (performance issues)

REFACTORING ==>								#
floats												#More than 10 float (prefer CSS grid)
font-sizes										#More than 10 font-size (prefer wrapping in few .CLASS { font-size: NUM })
font-faces										#More than 5 font-face (performance issues)

COMPATIBILITY ==>							#
non-link-hover								#With DOCTYPE strict, :hover only on <a> (IE<=7)
box-sizing										#box-sizing (problems with IE<=7)
adjoining-class								#.CLASS.CLASS2 (problems with IE<=6)
bulletproof-font-face					#@font-face { src: url(URL) }, first URL must append a dummy ?#iefix, to fix problems with IE<=8
star-property-hack						#*VAR: VAL (make it work only on IE<=7)
underscore-property-hack			#_VAR: VAL (make it work only on IE<=6)
gradients											#Use all browser extensions with gradient()
compatible-vendor-prefixes		#Use all browser extensions with any property that requires it
vendor-prefix									#Use non-browser extensions after browser extensions.
fallback-colors								#If using rgba(), hsl() or hsla(), use fallback rgb(), #NNN or color name (IE<=8)


GULP-CSSLINT([OBJ|FILE])		  #Fills in VINYLOBJ.csslint:
															#  - success BOOL: if errors or warnings
															#  - errorCount NUM: warnings or errors
															#  - results OBJ_ARR: file STR, error OBJ
															#  - opt OBJ
                              #OBJ|FILE are configuration
                              #Version 0.1.5
GULP-CSSLINT.reporter
(FUNC(VINYLOBJ))				      #Only fires FUNC
