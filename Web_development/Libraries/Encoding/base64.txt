STANDARDS ==>                     #  - standard: RFC 4648
                                  #  - base64url: RFC 4648
                                  #  - OpenPGP: RFC 4880
                                  #  - MIME: RFC 2045

GOAL ==>                          #Encode binary <-> ASCII

FORMAT ==>                        
                                  #  - 6 bits of binary <-> 8 bits of character:
                                  #     - among 64-charset: 
                                  #        - A-Z a-z 0-9 + /
                                  #        - (base64url only) A-Z a-z 0-9 - _
                                  #     - (MIME) chars outside charset are allowed, but ignored
                                  #     - padding:
                                  #        - 0-bits if binary less than 6 bits
                                  #        - [A][=]= at end if characters not multiple of 4
                                  #           - (base64url only) percent-encodes =
                                  #  - whitespaces:
                                  #     - not significant
                                  #     - (MIME, OpenPGP) adds CR+LF every 76 chars (max length)

TOOLS ==>                         #  - binary: base64 (GNU coreutils)
                                  #  - JavaScript (BOM): WINDOW.atob|btoa()
