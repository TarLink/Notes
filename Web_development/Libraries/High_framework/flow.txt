VAR: TYPE                         #TYPE can be:
function(...): TYPE { ... }       #  - any: anything can transtype to it, it can transtype to anything. 
                                  #    Should be avoided unless impossible.
                                  #  - mixed: anything can transtype to it, but can only transtype to mixed
                                  #    Should be avoided unless impossible.
                                  #  - boolean
                                  #  - number
                                  #  - string
                                  #  - void: undefined. Used as function return type
                                  #  - Object: any OBJ
                                  #    Should be avoided unless impossible.
                                  #  - CLASS: CLASS instance
                                  #  - Class<CLASS>: CLASS constructor
                                  #  - Array<TYPE>: means same type
                                  #  - Function: any function
                                  #  - ([VAR: TYPE]...) => TYPE: specific function type
                                  #Also can:
                                  #  - ?TYPE: can be null|undefined
                                  #  - typeof VAR: same TYPE
                                  #    If VAR is CLASS (not instance), means Class<CLASS>
                                  #Functions args:
                                  #  - can be called with more args, but not less (unless using ?TYPE or ...TYPE)

TRANSTYPE EXCEPTIONS ==>          #Allowed:
                                  #  - DATE -> NUM

REFINING TYPE ==>                 #  - Test on type (including combination with && || !), i.e.:
                                  #      typeof VAR === '...'
                                  #      VAR instanceof CLASS
                                  #      VAR ===|== null|undefined
                                  #    Modify TYPE any|mixed in the test bloc.
                                  #  - VAR = VAL: 
                                  #    - modify TYPE any|mixed, if originally declared with no assignment
                                  #    - can be re-assigned

TYPE INFERRENCE ==>               #Inferred in those cases:
                                  #  - VAL (not VAR)
                                  #  - [var] VAR = VAL, providing VAL TYPE known, and VAR TYPE unspecified
                                  #  - FUNC([...]), providing return type known
                                  #Otherwise TYPE 'any'
                                  #Includes interface files of JavaScript, BOM/DOM, Node, React
                                  #So annotation only needed:
                                  #  - argument type when it cannot be inferred from body
                                  #  - required on module.exports.VAR
