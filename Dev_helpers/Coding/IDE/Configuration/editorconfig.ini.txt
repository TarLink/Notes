.editorconfig           #File indicating how IDE, text editors should treat the file (if not specified, use editor def).
                        #Is INI format. Should be UTF8. Can use # COMMENT
                        #Properties are:
                        #  [FILES]
                        #  VAR = VAL
                        #  ...
                        #Where FILES uses globbing:
                        #  - including globstar **
                        #  - If only filename, will match only filename.
                        #  - Use brace expansion to specify several individual files
                        #Properties can be: 
                        #  - indent_style: "tab" or "space"
                        #  - indent_size: NUM
                        #  - end_of_line: "lf", "cr" or "crlf"
                        #  - charset: "latin1", "utf-8", "utf-16be|le"
                        #  - max_line_length: NUM
                        #  - trim_trailing_whitespace: "true" or "false"
                        #  - insert_final_newline: "true" or "false"
                        #  - root: "true" if root .editorconfig. There can be children .editorconfig in subfolders.
