                                  #  - length()->NUM: O(n)
                                  #  - get(INDEX)->VAL: O(n)

                                  #  - remove(INDEX): O(n)
                                  #  - slice(INDEX, LENGTH): O(1)
                                  #Pros:
                                  #  - persistent data structure
